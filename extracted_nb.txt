def count_data_items(filenames):
    """Count the total number of samples by parsing the counts embedded in TFRecord filenames."""
    n = [int(re.compile(r'-([0-9]*)\.').search(fn).group(1)) for fn in filenames]
    return np.sum(n)

## 6. Data Augmentation

--- CELL 19 (markdown) ---
Two augmentation methods are used during training:

1. **Geometric transforms** -- rotation, shear, zoom, and shift combined
   into a single affine matrix so only one interpolation pass is needed.

2. **Random erasing** (cutout) -- a random rectangle is zeroed out, which
   stops the model from relying on one specific region of the image.

Both run inside `tf.data.Dataset.map` on the TPU, so they overlap with
training and barely affect total runtime.

--- CELL 20 (code) ---
def get_mat(rotation, shear, height_zoom, width_zoom, height_shift, width_shift):
    """
    Build a single 3x3 affine matrix that encodes rotation, shear,
    anisotropic zoom, and translation. Composing these into one matrix
    means only one interpolation pass per image.
    """
    # convert rotation and shear angles from degrees to radians
    rotation    = math.pi * rotation / 180.0
    shear       = math.pi * shear   / 180.0

    one  = tf.constant([1], dtype='float32')
    zero = tf.constant([0], dtype='float32')

    # rotation matrix
    c1 = tf.math.cos(rotation)
    s1 = tf.math.sin(rotation)
    rotation_matrix = tf.reshape(
        tf.concat([c1, s1, zero, -s1, c1, zero, zero, zero, one], axis=0),
        [3, 3]
    )

    # shear matrix
    c2 = tf.math.cos(shear)
    s2 = tf.math.sin(shear)
    shear_matrix = tf.reshape(
        tf.concat([one, s2, zero, zero, c2, zero, zero, zero, one], axis=0),
        [3, 3]
    )

    # zoom matrix
    zoom_matrix = tf.reshape(
        tf.concat([one/height_zoom, zero, zero,
                   zero, one/width_zoom, zero,
                   zero, zero, one], axis=0),
        [3, 3]
    )

    # shift matrix
    shift_matrix = tf.reshape(
        tf.concat([one, zero, height_shift,
                   zero, one, width_shift,
                   zero, zero, one], axis=0),
        [3, 3]
    )

    return rotation_matrix @ shear_matrix @ zoom_matrix @ shift_matrix


def transform(image, label):
    """
    Apply a random affine transform to a single training image.
    Parameters are drawn from normal distributions scaled to
    produce mild-to-moderate distortions.
    """
    DIM  = IMAGE_SIZE[0]
    XDIM = DIM % 2   # correction for odd sizes like 331

    rot     = 15.0 * tf.random.normal([1], dtype='float32')
    shr     =  5.0 * tf.random.normal([1], dtype='float32')
    h_zoom  =  1.0 + tf.random.normal([1], dtype='float32') / 10.0
    w_zoom  =  1.0 + tf.random.normal([1], dtype='float32') / 10.0
    h_shift = 16.0 * tf.random.normal([1], dtype='float32')
    w_shift = 16.0 * tf.random.normal([1], dtype='float32')

    m = get_mat(rot, shr, h_zoom, w_zoom, h_shift, w_shift)

    # build the pixel coordinate grid centered at the image midpoint
    x   = tf.repeat(tf.range(DIM // 2, -DIM // 2, -1), DIM)
    y   = tf.tile(tf.range(-DIM // 2, DIM // 2), [DIM])
    z   = tf.ones([DIM * DIM], dtype='int32')
    idx = tf.stack([x, y, z])

    # multiply the coordinate grid by the combined affine matrix
    idx2 = tf.linalg.matmul(m, tf.cast(idx, dtype='float32'))
    idx2 = tf.cast(idx2, dtype='int32')

    # clip transformed coordinates to valid pixel indices and sample the image
    idx2 = tf.clip_by_value(idx2, -DIM // 2 + XDIM + 1, DIM // 2 - 1)
    idx3 = tf.stack([DIM // 2 - idx2[0], idx2[1] + DIM // 2])
    d    = tf.gather_nd(image, tf.transpose(idx3))

    return tf.reshape(d, [DIM, DIM, 3]), label


print('Geometric transform functions defined.')

def random_erasing(img, sl=0.1, sh=0.2, rl=0.4, p=0.3):
    """
    Random Erasing augmentation. With probability `p`, a rectangular
    patch of the image is replaced with zeros. The patch area is
    sampled between `sl` and `sh` fractions of the total image area,
    with aspect ratio bounded by `rl`.
    """
    if tf.random.uniform([]) > p:
        return img

    h = tf.shape(img)[0]
    w = tf.shape(img)[1]
    c = tf.shape(img)[2]
    origin_area = tf.cast(h * w, tf.float32)

    e_size_l = tf.cast(tf.round(tf.sqrt(origin_area * sl * rl)), tf.int32)
    e_size_h = tf.cast(tf.round(tf.sqrt(origin_area * sh / rl)), tf.int32)

    e_height_h = tf.minimum(e_size_h, h)
    e_width_h  = tf.minimum(e_size_h, w)

    erase_height = tf.random.uniform([], minval=e_size_l, maxval=e_height_h, dtype=tf.int32)
    erase_width  = tf.random.uniform([], minval=e_size_l, maxval=e_width_h,  dtype=tf.int32)

    erase_area = tf.zeros([erase_height, erase_width, c], dtype=img.dtype)

    pad_h      = h - erase_height
    pad_top    = tf.random.uniform([], minval=0, maxval=pad_h, dtype=tf.int32)
    pad_bottom = pad_h - pad_top

    pad_w     = w - erase_width
    pad_left  = tf.random.uniform([], minval=0, maxval=pad_w, dtype=tf.int32)
    pad_right = pad_w - pad_left

    mask = tf.pad(
        erase_area,
        [[pad_top, pad_bottom], [pad_left, pad_right], [0, 0]],
        constant_values=1
    )

    return img * mask


print('Random erasing function defined.')

## 7. Dataset Pipeline

--- CELL 23 (markdown) ---
Each TFRecord stores images as JPEG byte strings with either a class label
(train/val) or a string ID (test). The `tf.data` pipeline chains parsing,
decoding, augmentation, batching, and prefetching so data flows to the
TPU without stalls.

A few notes on the setup:

- Labels are one-hot encoded and paired with `categorical_crossentropy`.
- Ordering is turned off during training for speed, but turned on during
  validation and testing so predictions line up with the ground truth.
- Augmentation only runs on the training split.

--- CELL 24 (code) ---
def decode_image(image_data):
    """Decode a JPEG byte string into a float32 tensor normalized to the [0, 1] range."""
    image = tf.image.decode_jpeg(image_data, channels=3)
    image = tf.cast(image, tf.float32) / 255.0
    image = tf.reshape(image, [*IMAGE_SIZE, 3])
    return image


def read_labeled_tfrecord(example):
    """Parse a single labeled training or validation example from a TFRecord."""
    schema = {
        'image': tf.io.FixedLenFeature([], tf.string),
        'class': tf.io.FixedLenFeature([], tf.int64),
    }
    parsed = tf.io.parse_single_example(example, schema)
    image  = decode_image(parsed['image'])
    label  = tf.cast(parsed['class'], tf.int32)
    return image, label


def read_unlabeled_tfrecord(example):
    """Parse a single unlabeled test example from a TFRecord."""
    schema = {
        'image': tf.io.FixedLenFeature([], tf.string),
        'id':    tf.io.FixedLenFeature([], tf.string),
    }
    parsed = tf.io.parse_single_example(example, schema)
    image  = decode_image(parsed['image'])
    idnum  = parsed['id']
    return image, idnum


def onehot(image, label):
    """Convert an integer class label into a one-hot encoded vector of length 104."""
    return image, tf.one_hot(label, len(CLASSES))


def load_dataset(filenames, labeled=True, ordered=False):
    """Load a list of TFRecord files and parse each example into (image, label) pairs."""
    options = tf.data.Options()
    if not ordered:
        options.experimental_deterministic = False

    dataset = tf.data.TFRecordDataset(filenames, num_parallel_reads=AUTO)
    dataset = dataset.with_options(options)
    parser  = read_labeled_tfrecord if labeled else read_unlabeled_tfrecord
    dataset = dataset.map(parser, num_parallel_calls=AUTO)
    return dataset


print('Dataset reader functions defined.')

--- CELL 25 (code) ---
def get_training_dataset():
    """Assemble the training data pipeline: load, augment, one-hot encode, batch, and prefetch."""
    dataset = load_dataset(TRAINING_FILENAMES, labeled=True, ordered=False)
    dataset = dataset.map(transform, num_parallel_calls=AUTO)       # apply geometric augmentation
    dataset = dataset.map(lambda img, lbl: (random_erasing(img), lbl),  # apply random erasing
                          num_parallel_calls=AUTO)
    dataset = dataset.map(onehot, num_parallel_calls=AUTO)          # convert labels to one-hot vectors
    dataset = dataset.repeat()
    dataset = dataset.shuffle(2048)
    dataset = dataset.batch(BATCH_SIZE)
    dataset = dataset.prefetch(AUTO)
    return dataset


def get_validation_dataset(ordered=False):
    """Assemble the validation data pipeline without any augmentation."""
    dataset = load_dataset(VALIDATION_FILENAMES, labeled=True, ordered=ordered)
    dataset = dataset.map(onehot, num_parallel_calls=AUTO)
    dataset = dataset.batch(BATCH_SIZE)
    dataset = dataset.cache()
    dataset = dataset.prefetch(AUTO)
    return dataset


def get_test_dataset(ordered=False, augmented=False):
    """
    Build the test pipeline. When `augmented=True`, geometric
    transforms are applied for Test-Time Augmentation.
    """
    dataset = load_dataset(TEST_FILENAMES, labeled=False, ordered=ordered)
    if augmented:
        # apply the same geometric transforms used during training for Test-Time Augmentation
        dataset = dataset.map(lambda img, idnum: (transform(img, img)[0], idnum),
                              num_parallel_calls=AUTO)
    dataset = dataset.batch(BATCH_SIZE)
    dataset = dataset.prefetch(AUTO)
    return dataset


print('All dataset pipelines defined.')

def display_batch(dataset, title='Batch', rows=4, cols=5):
    """Display a grid of sample images from a batched dataset with their class names."""
    images, labels = next(iter(dataset.unbatch().batch(rows * cols)))

        if labels.numpy().ndim > 1:
            class_idx = np.argmax(labels[i].numpy())
        else:
            class_idx = labels[i].numpy()
        ax.set_title(CLASSES[class_idx], fontsize=8)
        ax.axis('off')

## 9. Learning Rate Schedule

--- CELL 31 (markdown) ---
When fine-tuning a pretrained model, the learning rate matters a lot.
Too high and you break the pretrained weights. Too low and nothing
moves. The schedule here ramps up from a small value to a peak over
the first five epochs, then decays exponentially. This is a common
pattern for transformer fine-tuning.

--- CELL 32 (code) ---
LR_START         = 0.00001
LR_MAX           = 0.00005 * strategy.num_replicas_in_sync
LR_MIN           = 0.00001
LR_RAMPUP_EPOCHS = 5
LR_SUSTAIN_EPOCHS = 0
LR_EXP_DECAY     = 0.8


def lrfn(epoch):
    """Compute the learning rate for a given epoch using a ramp-up, sustain, and decay schedule."""
    if epoch < LR_RAMPUP_EPOCHS:
        # linear ramp-up
        lr = (LR_MAX - LR_START) / LR_RAMPUP_EPOCHS * epoch + LR_START
    elif epoch < LR_RAMPUP_EPOCHS + LR_SUSTAIN_EPOCHS:
        # constant plateau
        lr = LR_MAX
    else:
        # exponential decay
        lr = (LR_MAX - LR_MIN) * LR_EXP_DECAY ** (
            epoch - LR_RAMPUP_EPOCHS - LR_SUSTAIN_EPOCHS
        ) + LR_MIN
    return lr


lr_callback = callbacks.LearningRateScheduler(lrfn, verbose=0)

# visualize the learning rate schedule across all epochs
rng = range(EPOCHS)
plt.figure(figsize=(10, 4))
plt.plot(rng, [lrfn(e) for e in rng], marker='o', markersize=3)
plt.xlabel('Epoch')
plt.ylabel('Learning Rate')
plt.title('Learning Rate Schedule')
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.show()

print(f'Peak LR: {LR_MAX:.6f}')

## 10. EfficientNetV2-L: Architecture and Training

--- CELL 34 (markdown) ---
EfficientNetV2-L is the large variant of Google's EfficientNetV2 family,
which was designed for both training speed and parameter efficiency. It
uses a combination of Fused-MBConv and standard MBConv blocks, with
progressive learning built into its training procedure. When pretrained
on ImageNet-21k and fine-tuned on a downstream task like flower
classification, it delivers strong accuracy.

The model is loaded without its classification head (`include_top=False`)
and a new dense layer with 104 outputs is appended. A global average
pooling layer sits between the feature extractor and the classifier to
reduce the spatial dimensions to a single feature vector.

Compilation details:

- Adam optimizer, default betas
- Categorical crossentropy (matches one-hot labels)
- Macro F1 as the tracked metric (same as competition scoring)

A checkpoint saves the best weights by validation F1. That way we always
run inference with the strongest epoch, not necessarily the last one.

--- CELL 35 (code) ---
with strategy.scope():

    backbone = tf.keras.applications.EfficientNetV2L(
        include_top=False,

        input_shape=(*IMAGE_SIZE, 3),
        pooling='avg'

    model = tf.keras.Sequential([
        backbone,
        L.Dense(len(CLASSES), activation='softmax')
    ])

    model.compile(
        optimizer='adam',
        loss='categorical_crossentropy',
        metrics=[tf.keras.metrics.F1Score(average='macro', name='f1_score')]

## 13. Test-Time Augmentation and Submission

--- CELL 46 (markdown) ---
Test-Time Augmentation (TTA) runs the model twice on each test image:
once on the original, once on an augmented copy. The softmax outputs
are averaged before argmax. This smooths out noise from any single
forward pass and usually bumps the F1 score by 0.5-1.5 points. On a
TPU the extra pass finishes fast, so there is no real downside.

--- CELL 47 (code) ---
# first forward pass: predict on the original, unmodified test images
test_ds_original = get_test_dataset(ordered=True, augmented=False)
preds_original   = model.predict(test_ds_original, verbose=0)

# second forward pass: predict on geometrically augmented test images
test_ds_augmented = get_test_dataset(ordered=True, augmented=True)
preds_augmented   = model.predict(test_ds_augmented, verbose=0)

# average the softmax probabilities from both passes, then select the class with the highest score
preds_avg = (preds_original + preds_augmented) / 2.0
predictions = np.argmax(preds_avg, axis=1)

print(f'Predictions shape: {predictions.shape}')
print(f'Unique classes predicted: {len(np.unique(predictions))}')

--- CELL 48 (code) ---
# extract the test image identifiers in the same order used during prediction
test_ids_ds = test_ds_original.map(lambda image, idnum: idnum).unbatch()
test_ids    = next(iter(test_ids_ds.batch(NUM_TEST_IMAGES))).numpy().astype('U')

# assemble the submission dataframe with columns 'id' and 'label'
submission = pd.DataFrame({'id': test_ids, 'label': predictions})

print(f'Submission shape: {submission.shape}')
print(f'Label range:      [{predictions.min()}, {predictions.max()}]')
print(f'Null values:      {submission.isnull().sum().sum()}')
