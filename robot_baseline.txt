## Subtask 1: Robust Readability Metric (RRM)

Deterministic computation:
```
RRM = clip(206.835 - 1.015 * (W / S) - 84.6 * (SYL / W), 0, 100)
```

The key challenge is accurate syllable counting. We use a vowel-group heuristic with corrections for silent-e, common suffixes, etc.

---CELL---

def count_syllables(word):
    """Count syllables using vowel-group heuristic with corrections."""
    word = word.lower().strip()
    if not word:
        return 0
    # Remove non-alpha
    word = re.sub(r'[^a-z]', '', word)
    if not word:
        return 0
    if len(word) <= 2:
        return 1

    vowels = 'aeiouy'
    count = 0
    prev_vowel = False
    for char in word:
        is_vowel = char in vowels
        if is_vowel and not prev_vowel:
            count += 1
        prev_vowel = is_vowel

    # Silent e at end
    if word.endswith('e') and not word.endswith('le'):
        count -= 1
    # -ed ending (usually silent unless preceded by t or d)
    if word.endswith('ed') and len(word) > 3 and word[-3] not in 'td':
        count -= 1

    return max(count, 1)


def count_sentences(text):
    """Count sentences by splitting on sentence-ending punctuation."""
    sentences = re.split(r'[.!?]+', text.strip())
    sentences = [s.strip() for s in sentences if s.strip()]
    return max(len(sentences), 1)


def count_words(text):
    """Count words."""
    words = text.split()
    return max(len(words), 1)


def compute_rrm(text):
    """Compute Robust Readability Metric."""
    if not text or not text.strip():
        return 50.0

    words = text.split()
    W = max(len(words), 1)
    S = count_sentences(text)
    SYL = sum(count_syllables(w) for w in words)

    rrm = 206.835 - 1.015 * (W / S) - 84.6 * (SYL / W)
    return np.clip(rrm, 0.0, 100.0)

def extract_text_features(df):
    """Extract handcrafted text statistics features."""
    feats = pd.DataFrame(index=df.index)
    text = df['text_content']

    feats['n_chars'] = text.str.len()
    feats['n_words'] = text.apply(lambda x: len(x.split()))
    feats['n_sentences'] = text.apply(count_sentences)
    feats['n_paragraphs'] = text.apply(lambda x: len(x.strip().split('\n')))
    feats['avg_word_len'] = text.apply(lambda x: np.mean([len(w) for w in x.split()]) if x.split() else 0)
    feats['avg_sent_len'] = feats['n_words'] / feats['n_sentences']
    feats['avg_para_len'] = feats['n_words'] / feats['n_paragraphs']
    feats['unique_word_ratio'] = text.apply(
        lambda x: len(set(x.lower().split())) / max(len(x.lower().split()), 1)
    )
    feats['std_word_len'] = text.apply(
        lambda x: np.std([len(w) for w in x.split()]) if len(x.split()) > 1 else 0
    )

    # Punctuation features
    feats['n_commas'] = text.str.count(',')
    feats['n_periods'] = text.str.count(r'\.')
    feats['n_exclamation'] = text.str.count('!')
    feats['n_question'] = text.str.count(r'\?')
    feats['punct_ratio'] = (feats['n_commas'] + feats['n_periods'] +
                            feats['n_exclamation'] + feats['n_question']) / feats['n_words']

    # Capitalization
    feats['upper_ratio'] = text.apply(lambda x: sum(1 for c in x if c.isupper()) / max(len(x), 1))

    # Syllable-based
    feats['avg_syllables'] = text.apply(
        lambda x: np.mean([count_syllables(w) for w in x.split()]) if x.split() else 0
    )
    feats['rrm'] = text.apply(compute_rrm)

    # Sentence length variance (burstiness)
    def sent_len_std(t):
        sents = [s.strip() for s in re.split(r'[.!?]+', t.strip()) if s.strip()]
        if len(sents) < 2:
            return 0.0
        lens = [len(s.split()) for s in sents]
        return np.std(lens)

    feats['sent_len_std'] = text.apply(sent_len_std)

    return feats

## Subtask 3: Human vs AI Detection (Unsupervised)

No labels are available. Strategy:
1. Extract text-quality features that differentiate human vs AI writing
2. Use KMeans clustering (k=2) on the combined train+test data
3. Assign cluster labels based on AI text characteristics (lower burstiness, more uniform sentence length, higher vocabulary consistency)

---CELL---

def extract_ai_detection_features(df):
    """Features specifically designed to distinguish AI from human text."""
    feats = pd.DataFrame(index=df.index)
    text = df['text_content']

    # Sentence-level statistics (AI text is more uniform)
    def sentence_stats(t):
        sents = [s.strip() for s in re.split(r'[.!?]+', t.strip()) if s.strip()]
        if len(sents) < 2:
            return 0.0, 0.0, 0.0
        lens = [len(s.split()) for s in sents]
        return np.std(lens), np.mean(lens), np.max(lens) - np.min(lens)

    stats = text.apply(sentence_stats)
    feats['sent_len_std'] = stats.apply(lambda x: x[0])
    feats['sent_len_mean'] = stats.apply(lambda x: x[1])
    feats['sent_len_range'] = stats.apply(lambda x: x[2])

    # Burstiness: coefficient of variation of sentence lengths
    feats['sent_len_cv'] = feats['sent_len_std'] / (feats['sent_len_mean'] + 1e-8)

    # Vocabulary richness
    feats['unique_ratio'] = text.apply(
        lambda x: len(set(x.lower().split())) / max(len(x.lower().split()), 1)
    )

    # Hapax legomena ratio (words appearing only once)
    def hapax_ratio(t):
        words = t.lower().split()
        if not words:
            return 0.0
        from collections import Counter
        freq = Counter(words)
        hapax = sum(1 for w, c in freq.items() if c == 1)
        return hapax / len(words)
    feats['hapax_ratio'] = text.apply(hapax_ratio)

    # Paragraph structure
    feats['n_paragraphs'] = text.apply(lambda x: len(x.strip().split('\n')))
    feats['words_per_para'] = text.apply(lambda x: len(x.split())) / feats['n_paragraphs']

    # Punctuation patterns
    feats['comma_rate'] = text.str.count(',') / text.apply(lambda x: max(len(x.split()), 1))
    feats['period_rate'] = text.str.count(r'\.') / text.apply(lambda x: max(len(x.split()), 1))

    # Word length distribution (AI tends to be more consistent)
    def word_len_stats(t):
        words = t.split()
        if len(words) < 2:
            return 0.0, 0.0
        lens = [len(w) for w in words]
        return np.std(lens), np.median(lens)
    wstats = text.apply(word_len_stats)
    feats['word_len_std'] = wstats.apply(lambda x: x[0])
    feats['word_len_median'] = wstats.apply(lambda x: x[1])

    # Transition words (human text tends to use more)
    transitions = r'\b(however|therefore|moreover|furthermore|nevertheless|although|meanwhile|consequently|additionally|specifically)\b'
    feats['transition_count'] = text.str.lower().str.count(transitions)
    feats['transition_rate'] = feats['transition_count'] / text.apply(lambda x: max(len(x.split()), 1))

    # First-person pronouns (human essays use more personal language)
    first_person = r'\b(I|my|me|myself|mine|we|our|us|ourselves)\b'
    feats['first_person_count'] = text.str.count(first_person)
    feats['first_person_rate'] = feats['first_person_count'] / text.apply(lambda x: max(len(x.split()), 1))

    # Text length
    feats['n_words'] = text.apply(lambda x: len(x.split()))
    feats['n_chars'] = text.str.len()

    return feats

## Summary & Potential Improvements

**Baseline approach:**
- **Subtask 1 (RRM):** Deterministic formula with vowel-group syllable counting
- **Subtask 2 (Sentiment):** TF-IDF (word + char n-grams) + handcrafted features → Ensemble (Ridge + SVR + GBR)
- **Subtask 3 (AI detection):** Text statistics features → KMeans clustering (k=2) with heuristic cluster assignment

**Ideas for improvement:**
1. Use CMU Pronouncing Dictionary (nltk) for more accurate syllable counting in subtask 1
2. Try transformer-based embeddings (sentence-transformers) for sentiment and AI detection
3. Use Gaussian Mixture Models or DBSCAN instead of KMeans for subtask 3
4. Add more linguistic features: POS tag distributions, dependency parse features
5. Try semi-supervised learning for subtask 3 if any weak labels can be derived
6. Use cross-validated stacking instead of simple voting ensemble for subtask 2
7. Feature selection to reduce overfitting on the TF-IDF features
